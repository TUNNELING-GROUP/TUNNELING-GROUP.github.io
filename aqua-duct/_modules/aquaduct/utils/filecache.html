[docs]    def __call__(self, func, *args, **kwargs):
        def wrapper(*args, **kwargs):
            key = md5(','.join(map(str, args)) + '&' + ','.join(map(lambda kv: ':'.join(map(str, kv)), kwargs.iteritems()))).hexdigest()

            logger.debug('Looking for cache key {}'.format(key))

            try:
                coords = np.load(self.db_dir + key + ".npy")
                print "Data fetched"
                return coords
            except IOError:
                coords = func(*args, **kwargs)

                np.save(self.db_dir + key, coords)

                logger.debug("New key added to cache.")

                return coords


        return wrapper
